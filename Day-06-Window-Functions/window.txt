WITH RankedSales AS (
    SELECT 
        id,
        employee_name,
        department,
        current_sales_volume,
        DENSE_RANK() OVER (
            PARTITION BY department            
            ORDER BY current_sales_volume DESC
        ) as sales_rank
    FROM department_sales_targets
)
SELECT id, employee_name, department, current_sales_volume
FROM RankedSales
WHERE sales_rank = 1;

2.
SELECT 
    id,
    employee_name,
    department,
    current_sales_volume,
    SUM(current_sales_volume) OVER (
        ORDER BY current_sales_volume DESC
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_sales
FROM department_sales_targets;

3.
SELECT 
    revenue_year,
    revenue_month,
    total_amount AS current_month_revenue,
    
    -- 1. Grab the revenue from the previous month row
    LAG(total_amount, 1) OVER (
        ORDER BY revenue_year, revenue_month
    ) AS previous_month_revenue,
    
    -- 2. Calculate the raw difference between this month and last month
    (total_amount - LAG(total_amount, 1) OVER (
        ORDER BY revenue_year, revenue_month
    )) AS net_growth
    
FROM monthly_revenue;
