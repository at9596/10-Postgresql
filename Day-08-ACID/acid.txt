2. Concurrency:
window : 1
    BEGIN;

    -- Select Alice's record and place an exclusive lock on her row
    SELECT balance FROM accounts WHERE id = 1 FOR UPDATE;
window 2(Simultaneous Execution)

BEGIN;

-- Try to modify the exact same row that Connection 1 locked
UPDATE accounts SET balance = balance - 50.00 WHERE id = 1;

-- [THE TERMINAL WAITING FOR FIRST TRANSACTION COMPLITATION]: 
-- The cursor blinks and hangs. Connection 2 is now waiting in line.

Back inside Window 1:

-- Complete the work and commit
UPDATE accounts SET balance = balance - 100.00 WHERE id = 1;
COMMIT; 

-- The moment this COMMIT executes, Connection 1 drops its lock. 
-- Connection Window #2 instantly unfreezes and executes its update safely!
